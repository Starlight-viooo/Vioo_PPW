import pandas as pd

def clean_text(text):
    # 1. Mengubah teks menjadi huruf kecil (lowercase)
    text = text.lower()
    # 2. Menghilangkan angka
    text = re.sub(r"\d+", "", text)
    # 3. Menghilangkan tanda baca (punctuation)
    text = text.translate(str.maketrans("","",string.punctuation))
    # 4. Menghilangkan spasi putih berlebih
    text = text.strip()
    return text

from sklearn.feature_extraction.text import TfidfVectorizer
import pandas as pd

# --- 1. Memuat Data yang Sudah Diproses ---
NAMA_FILE_PROCESSED = 'berita_tempo_unik.csv'

try:
    df = pd.read_csv(NAMA_FILE_PROCESSED)
    # Ambil kolom 'isi_final' yang berisi teks bersih dan gabungkan menjadi sebuah list
    # Kita juga perlu menangani baris yang mungkin kosong (NaN) dengan mengubahnya menjadi string kosong
    corpus = df['kata_unik_final'].fillna('').tolist()
    print(f"✅ Berhasil memuat {len(corpus)} dokumen dari '{NAMA_FILE_PROCESSED}' untuk perhitungan TF-IDF.")

except FileNotFoundError:
    print(f"❌ ERROR: File '{NAMA_FILE_PROCESSED}' tidak ditemukan. Pastikan Anda sudah menjalankan sel preprocessing sebelumnya.")
    corpus = [] # Buat list kosong agar tidak error

# --- 2. Menjalankan TF-IDF ---
if corpus:
    # Inisialisasi TfidfVectorizer
    vectorizer = TfidfVectorizer()

    # Lakukan fit_transform untuk mempelajari vocabulary dan menghitung bobot TF-IDF
    X = vectorizer.fit_transform(corpus)

    # --- 3. Menampilkan Hasil ---
    print("\n--- Hasil Perhitungan TF-IDF ---")

    # Tampilkan daftar kata (fitur) yang berhasil diidentifikasi
    print("Contoh beberapa fitur (kata) yang ditemukan:")
    print(vectorizer.get_feature_names_out()[:20]) # Tampilkan 20 kata pertama

    # Tampilkan dimensi dari matriks TF-IDF yang dihasilkan
    print("\nDimensi Matriks TF-IDF (dokumen, fitur):")
    print(X.shape)

    # Menampilkan matriks dalam bentuk DataFrame agar mudah dibaca (opsional, bisa sangat besar)
    print("\n--- Contoh Matriks TF-IDF (5 dokumen pertama, 10 fitur pertama) ---")
    df_tfidf = pd.DataFrame(X.toarray()[:5, :10], columns=vectorizer.get_feature_names_out()[:10])
    display(df_tfidf)

from gensim.models import Word2Vec
import numpy as np
import pandas as pd

# --- 1. Memuat Data Unik ---
NAMA_FILE_UNIK = 'berita_tempo_unik.csv'

try:
    df_processed = pd.read_csv(NAMA_FILE_UNIK)
    print(f"✅ Berhasil memuat data dari '{NAMA_FILE_UNIK}'")
except FileNotFoundError:
    print(f"❌ ERROR: File '{NAMA_FILE_UNIK}' tidak ditemukan. Pastikan Anda sudah menjalankan sel preprocessing sebelumnya.")
    raise # Hentikan eksekusi jika file tidak ada

# --- 2. Mempersiapkan Data untuk Model ---

# Nama kolom yang berisi teks bersih (kata-kata unik yang digabung)
kolom_teks = 'kata_unik_final'

if kolom_teks not in df_processed.columns:
    print(f"❌ ERROR: Kolom '{kolom_teks}' tidak ditemukan dalam dataframe.")
else:
    # Word2Vec membutuhkan input berupa list dari list token.
    # Kita ubah setiap kalimat di kolom 'kata_unik_final' menjadi list token dengan .split()
    sentences = df_processed[kolom_teks].fillna('').apply(lambda x: x.split()).tolist()

    print("\n=== Proses Pelatihan Word Embedding (Word2Vec) ===")

    # --- 3. Melatih Model Word2Vec ---
    # vector_size: dimensi vektor untuk setiap kata
    # window: ukuran jendela konteks (kata-kata tetangga)
    # min_count: abaikan kata yang muncul kurang dari nilai ini
    word2vec_model = Word2Vec(sentences=sentences, vector_size=100, window=5, min_count=1, workers=4)
    print("✅ Model Word2Vec berhasil dilatih.")

    # --- 4. Membuat Vektor Dokumen ---

    # Fungsi untuk mendapatkan vektor dokumen (rata-rata vektor kata)
    def get_document_vector(tokens, model):
        # Ambil vektor untuk setiap token yang ada di dalam vocabulary model
        vectors = [model.wv[token] for token in tokens if token in model.wv]
        if vectors:
            # Rata-ratakan semua vektor kata untuk mendapatkan satu vektor dokumen
            return np.mean(vectors, axis=0)
        else:
            # Kembalikan vektor nol jika tidak ada kata yang dikenali model
            return np.zeros(model.wv.vector_size)

    # Terapkan fungsi ke 'sentences' untuk mendapatkan vektor setiap berita
    df_processed['document_vector'] = [get_document_vector(doc, word2vec_model) for doc in sentences]

    # --- 5. Menampilkan Hasil ---

    # Buat DataFrame dari vektor dokumen agar mudah dibaca
    document_vector_df = pd.DataFrame(df_processed['document_vector'].tolist(),
                                      columns=[f'dim_{i}' for i in range(word2vec_model.wv.vector_size)])

    print("\n--- Contoh Vektor Dokumen (5 Berita Pertama) ---")
    display(document_vector_df.head())

    print("\n✨ Proses Word Embedding selesai.")
